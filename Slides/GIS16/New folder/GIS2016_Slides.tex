\section{Definitions}
\frame{\frametitle{Outline}\tableofcontents}

\begin{frame}\frametitle{Definitions}
\vspace{-0.2in}
\only<1->{
	\begin{block}{Ride Request}
		A ride request \textit{r} is represented as $\left\langle s, e, w, \epsilon, \psi \right\rangle$ where:
		\begin{itemize}
			\item \textit{s}: pickup point
			\item \textit{e}: dropoff point
			\item \textit{w}: max wait time
			\item \textit{$\epsilon$}: max detour
			\item \textit{$\psi$}: rider's profile
		\end{itemize}			 
	\end{block}
}
\only<2>{
	\begin{alertblock}{Driver}
		A driver \textit{d} is represented as $\left\langle \mathcal{R_d}, n, \theta \right\rangle$ such that:
		\begin{itemize}
			\item $\mathcal{R_d}$: list of assigned requests
			\item \textit{n}: max simultaneous passengers 
			\item $\theta$: driver's profile
		\end{itemize}			 
	\end{alertblock}
}
\end{frame}

\begin{frame}\frametitle{Definitions}
\begin{exampleblock}{Schedule}
For a set $\mathcal{R_d}$ with $n$ requests, a schedule $\pi = \left\langle x_1, x_2, ..., x_{2n} \right\rangle$ is an ordered set of pick-up and drop-off points of the requests in $L$.
\end{exampleblock}

\only<2>{
We call $\pi$ a \textit{feasible} schedule if it satisfies these constraints:
\begin{itemize}
\item for every $r \in \mathcal{R_d}$, $s_r$ precedes $e_r$ in $\pi$
\item the rider's waiting time and detour
\item the driver's capacity
\end{itemize}
}

\section{Pricing Model}
\frame{\frametitle{Outline}\tableofcontents[currentsection]}

\begin{frame}\frametitle{Fair Pricing}
For every pricing model:
\begin{itemize}
\item How much should the rider pay?
\item How much should the driver be compensated?
\item What's the revenue of the ride-sharing platform?
\end{itemize}
\vspace{0.5in}
\only<2->{
In a \textit{fair} system:
\begin{itemize}
\item the rider should receive a \textit{discount} proportional the the detour incurred to his trip
\item a driver's compensation should increase proportional to the distance of his trip
\end{itemize}
}
\end{frame}

\begin{frame}\frametitle{Rider's Fare}
for every request \textit{r}:
\begin{itemize}
\item<1-> $\phi_r$: shortest path between $s_r$ and $e_r$
\item<2-> $F: \mathbb{R}_{+}  \rightarrow \$ $ such that $F(\phi_r)$ is the default fare of a ride
\end{itemize}
\begin{columns}
\column{0.5\textwidth}
\begin{itemize}
\item<3-> $\hat{\phi}_r$: actual trip of \textit{r}
\item<3-> $\Delta_r = \hat{\phi}_r - \phi_r$
\item<4-> $\psi: \mathbb{R}_{+} \rightarrow \left[ 0, 1 \right] $ specifies the discount for $\Delta d_r \in \mathbb{R}_{+}$
\end{itemize}
\vspace{0.25in}
\only<5->{
\begin{block}{}
\begin{equation*}
fare(r) = F(\phi_r) \psi_r(\Delta_r)
\end{equation*}
\end{block}
}

\column{0.5\textwidth}
\only<5->{
\begin{figure}
	\centering
    \includegraphics[width = 0.95\columnwidth]{rider.eps}
    \label{fig:quality}
\end{figure}
}
\end{columns}
\end{frame}

\begin{frame}\frametitle{Driver's Cost}
\vspace{-0.15in}
for every driver \textit{v}:
\begin{itemize}
\item $\theta: \mathbb{R}_{+}  \rightarrow \$ $ specifies the monetary cost of \textit{d} driving for a duration $\omega \in \mathbb{R}_{+}$
\begin{itemize}
\item<2> $\theta: \mathbb{N} \times \mathbb{R}_{+}  \rightarrow \$$
\end{itemize}
\item<2-> $d$'s reported profile: $\hat{\theta_d}$
\end{itemize}
\begin{columns}
\column{0.60\textwidth}
\vspace{-0.1in}
\only<4->{
\begin{exampleblock}{}
\begin{equation*}
cost(\pi_d, \hat{\theta_d}) = \int_{start_{\pi_d}}^{end_{\pi_d}} I\left( \pi_d^t \neq \left\langle \right\rangle\right).\hat{\theta_d}(t)dt
\end{equation*}
\begin{itemize}
\item $I()$: indicator function
\item $\pi_d^t$: driver's schedule at \textit{t}. 
\item $start_{\pi_d}$: first pickup time of $S_v$
\item $end_{\pi_d}$: last dropoff time of $S_v$
\item $d(t)$ traveled distance of \textit{v} at \textit{t}
\end{itemize}
\end{exampleblock}
}
\column{0.40\textwidth}
\only<3->{
\begin{figure}
	\centering
    \includegraphics[width = 0.95\columnwidth]{driver.eps}
    \label{fig:quality}
\end{figure}
}
\end{columns}
\end{frame}

\begin{frame}\frametitle{Driver's Income \& Utility}
\begin{itemize}
\item Payments \& Income:
\begin{align*}
payment(\pi_d, \hat{\theta_d}) &= \sum_{r \in \pi_d} \rho_r\\
income(\pi_d, \hat{\theta_d}) &= \sum_{r \in \pi_d} fare(r) - payment(\pi_d, \hat{\theta_d})
\end{align*}
\item<2-> Utility:
\only<2->{
\begin{equation*}
u(\pi_d, \hat{\theta_d}, \theta_d) = income(\pi_d, \hat{\theta_d}) - cost(\pi_d, \theta_d)
\end{equation*}
}
\end{itemize}
\end{frame}

\begin{frame}\frametitle{Revenue}
\begin{exampleblock}{Revenue}
Given the matching $M(\mathcal{D}, \mathcal{R})$ between the set of drivers $\mathcal{D}$ and requests $\mathcal{R}$, the revenue of the platform provider is
\begin{equation*}
revenue(M(\mathcal{D}, \mathcal{R})) = \sum_{d \in \mathcal{D}} payments(\pi_d, \hat{\theta_d})
\end{equation*}
\end{exampleblock}
\end{frame}
